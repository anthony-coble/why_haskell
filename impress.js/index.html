// Fibonacci numbers, imperative style
int fibonacci(int iterations)
{
  int first = 0, second = 1; // seed values

  for (int i = 0; i < iterations; ++i) {
    int sum = first + second;
    first = second;
    second = sum;
  }

  return first;
}

int main()
{
  std::cout << fibonacci(10) << "\n";
  return 0;
}
